83      } else {

85      }
86      if (!DataServerUri.isEmpty()) {
87        this.dataStoreServer =
88            HttpInvokerUtils.createStreamSupportingServiceStub(
89                IDataStoreServerApi.class, DataServerUri + IDataStoreServerApi.SERVICE_URL, 10000);
90      } else {
91        this.dataStoreServer = null;
92      }
93  
94      this.setCredentials(user, password);

97    private static Path getTopDirectory(Path path) {
98      Path currentPath = Paths.get(path.toString());
99      Path parentPath;
100     while (currentPath.getParent() != null) {
101       parentPath = currentPath.getParent();
102       currentPath = parentPath;
103     }
104     return currentPath;
105   }

137   /**
138    * Downloads the files that the user requested checks whether any filtering option (suffix or
139    * regex) has been passed and applies filtering if needed
140    *
141    * @param commandLineParameters
142    * @param qbicDataDownloader
143    * @throws IOException
144    */
145   public void downloadRequestedFilesOfDatasets(
146       PostmanCommandLineOptions commandLineParameters, QbicDataDownloader qbicDataDownloader)
147       throws IOException {
148     QbicDataFinder qbicDataFinder =
149         new QbicDataFinder(applicationServer, dataStoreServer, sessionToken, filterType);
150 
151     LOG.info(
152         String.format(
153             "%s provided openBIS identifiers have been found: %s",
154             commandLineParameters.ids.size(), commandLineParameters.ids.toString()));
155 
156     // a suffix was provided -> only download files which contain the suffix string
157     if (!commandLineParameters.suffixes.isEmpty()) {
158       for (String ident : commandLineParameters.ids) {
159         LOG.info(String.format("Downloading filtered files for provided identifier %s", ident));
160         List<Map<String, List<DataSetFile>>> foundSuffixFilteredIDs =
161             qbicDataFinder.findAllSuffixFilteredIDs(ident, commandLineParameters.suffixes);
162 
163         LOG.info(String.format("Number of files found: %s", countDatasets(foundSuffixFilteredIDs)));
164 
165         downloadFilesFilteredByIDs(ident, foundSuffixFilteredIDs);
166       }
167       // a regex pattern was provided -> only download files which contain the regex pattern
168     } else if (!commandLineParameters.regexPatterns.isEmpty()) {
169       for (String ident : commandLineParameters.ids) {
170         LOG.info(String.format("Downloading files for provided identifier %s", ident));
171         List<DataSetFile> foundRegexFilteredIDs =
172             qbicDataFinder.findAllRegexFilteredIDs(ident, commandLineParameters.regexPatterns);
173 
174         LOG.info(String.format("Number of files found: %s", foundRegexFilteredIDs.size()));
175 
176         //downloadFilesFilteredByIDs(ident, foundRegexFilteredIDs);
177       }
178     } else {
179       // no suffix or regex was supplied -> download all datasets
180       for (String ident : commandLineParameters.ids) {
181         LOG.info(String.format("Downloading files for provided identifier %s", ident));
182         Map<String, List<DataSet>> foundDataSets =
183             qbicDataFinder.findAllDatasetsRecursive(ident);
184 
185         LOG.info(String.format("Number of datasets found: %s", countDatasets(foundDataSets)));
186 
187         if (foundDataSets.size() > 0) {
188           LOG.info("Initialize download ...");
189           int datasetDownloadReturnCode = -1;
190           try {
191             // for the sample code and aggregates datasets per sample code
192             List<Map<String, List<DataSet>>> datasets = new ArrayList<>();
193             datasets.add(foundDataSets);
194             datasetDownloadReturnCode =
195                 qbicDataDownloader.downloadDataset(datasets);
196           } catch (NullPointerException e) {
197             LOG.error(
198                 "Datasets were found by the application server, but could not be found on the datastore server for "
199                     + ident
200                     + "."
201                     + " Try to supply the correct datastore server using a config file!");
202           }
203 
204           if (datasetDownloadReturnCode != 0) {
205             LOG.error("Error while downloading dataset: " + ident);
206           } else {
207             LOG.info("Download successfully finished.");
208           }
209 
210         } else {
211           LOG.info("Nothing to download.");
212         }
213       }
214     }
215   }

217   private <T> Integer countDatasets(List<Map<String, List<T>>> datasetsPerSampleCode) {
218     int sum = 0;
219     for (Map<String, List<T>> entry : datasetsPerSampleCode){
220       for (String sampleCode : entry.keySet()) {
221         sum += entry.get(sampleCode).size();
222       }
223     }
224     return sum;
225   }

227   private static <T> int countDatasets(Map<String, List<T>> datasetsPerSampleCode) {
228     return datasetsPerSampleCode.values().stream().mapToInt(List::size).sum();
229   }

233    * Downloads all IDs which were previously filtered by either suffixes or regexes
234    *
235    * @param ident
236    * @param foundFilteredDatasets
237    * @throws IOException
238    */
239   private void downloadFilesFilteredByIDs(String ident, List<Map<String, List<DataSetFile>>> foundFilteredDatasets)
240       throws IOException {
241     if (foundFilteredDatasets.size() > 0) {
242       LOG.info("Initialize download ...");
243       int filesDownloadReturnCode = -1;
244       try {
245         for (Map<String, List<DataSetFile>> filesPerSample : foundFilteredDatasets){
246           for (String sampleCode : filesPerSample.keySet()) {
247             List<DataSetFile> filteredDataSetFiles = withoutDirectories(filesPerSample.get(sampleCode));
248             final DownloadRequest downloadRequest = new DownloadRequest(filteredDataSetFiles,
249                 sampleCode);
250             filesDownloadReturnCode = downloadFiles(downloadRequest);
251           }
252         }
253       } catch (NullPointerException e) {
254         LOG.error(
255             "Datasets were found by the application server, but could not be found on the datastore server for "
256                 + ident
257                 + "."
258                 + " Try to supply the correct datastore server using a config file!");
259       }
260       if (filesDownloadReturnCode != 0) {
261         LOG.error("Error while downloading dataset: " + ident);
262       } else {
263         LOG.info("Download successfully finished");
264       }
265 
266     } else {
267       LOG.info("Nothing to download.");
268     }
269   }

284   private void downloadDataset(Map<String, List<DataSet>> dataSetsPerSample) {
285 
286     for (Entry<String, List<DataSet>> entry : dataSetsPerSample.entrySet()) {
287       String sampleCode = entry.getKey();
288       List<DataSet> sampleDatasets = entry.getValue();
289       for (DataSet sampleDataset : sampleDatasets) {
290         DataSetPermId permID = sampleDataset.getPermId();
291         DataSetFileSearchCriteria criteria = new DataSetFileSearchCriteria();
292         criteria.withDataSet().withCode().thatEquals(permID.getPermId());
293         SearchResult<DataSetFile> result =
294             this.dataStoreServer.searchFiles(sessionToken, criteria, new DataSetFileFetchOptions());
295         List<DataSetFile> filteredDataSetFiles = withoutDirectories(result.getObjects());
296         final DownloadRequest downloadRequest = new DownloadRequest(filteredDataSetFiles,
297             sampleCode, DEFAULT_DOWNLOAD_ATTEMPTS);
298         downloadFiles(downloadRequest);
299         //
300       }
301     }
302   }

304   private static List<DataSetFile> withoutDirectories(List<DataSetFile> dataSetFiles) {
305     Predicate<DataSetFile> notADirectory = dataSetFile -> !dataSetFile.isDirectory();
306     return dataSetFiles.stream()
307         .filter(notADirectory)
308         .collect(Collectors.toList());
309   }

311   private void downloadFile(DataSetFile dataSetFile, Path prefix) throws IOException {
312     DataSetFileDownloadOptions options = new DataSetFileDownloadOptions();
313     options.setRecursive(false);
314     IDataSetFileId fileId = dataSetFile.getPermId();
315     InputStream stream =
316         this.dataStoreServer.downloadFiles(
317             sessionToken, Collections.singletonList(fileId), options);
318     DataSetFileDownloadReader reader = new DataSetFileDownloadReader(stream);
319     DataSetFileDownload file;
320 
321     while ((file = reader.read()) != null) {
322       InputStream initialStream = file.getInputStream();
323       CheckedInputStream checkedInputStream = new CheckedInputStream(initialStream, new CRC32());
324       if (file.getDataSetFile().getFileLength() > 0) {
325         final Path filePath = determineFinalPathFromDataset(file.getDataSetFile());
326         File newFile =
327             new File(System.getProperty("user.dir") +
328                   File.separator +
329                   prefix.toString() + File.separator +
330                   filePath.toString());
331         newFile.getParentFile().mkdirs();
332         OutputStream os = new FileOutputStream(newFile);
333         ProgressBar progressBar =
334             new ProgressBar(
335                 filePath.getFileName().toString(), file.getDataSetFile().getFileLength());
336         int bufferSize =
337             (file.getDataSetFile().getFileLength() < defaultBufferSize)
338                 ? (int) file.getDataSetFile().getFileLength()
339                 : defaultBufferSize;
340         byte[] buffer = new byte[bufferSize];
341         int bytesRead;
342         // read from is to buffer
343         while ((bytesRead = checkedInputStream.read(buffer)) != -1) {
344           progressBar.updateProgress(bufferSize);
345           os.write(buffer, 0, bytesRead);
346           os.flush();
347         }
348         System.out.print("\n");
349         validateChecksum(
350             Long.toHexString(checkedInputStream.getChecksum().getValue()), dataSetFile);
351         initialStream.close();
352         // flush OutputStream to write any buffered data to file
353         os.flush();
354         os.close();
355       }
356     }
357   }

359   private void validateChecksum(String computedChecksumHex, DataSetFile dataSetFile) {
360     String expectedChecksum = Integer.toHexString(dataSetFile.getChecksumCRC32());
361     try {
362       if (computedChecksumHex.equals(expectedChecksum)) {
363         checksumReporter.reportMatchingChecksum(
364             expectedChecksum,
365             computedChecksumHex,
366             Paths.get(dataSetFile.getPath()).toUri().toURL());
367       } else {
368         checksumReporter.reportMismatchingChecksum(
369             expectedChecksum,
370             computedChecksumHex,
371             Paths.get(dataSetFile.getPath()).toUri().toURL());
372       }
373     } catch (MalformedURLException e) {
374       LOG.error(e);
375     }
376   }

378   private Path determineFinalPathFromDataset(DataSetFile file) {
379     Path finalPath;
380     if (conservePaths) {
381       finalPath = Paths.get(file.getPath());
382       // drop top parent directory name in the openBIS DSS (usually "/origin")
383       Path topDirectory = getTopDirectory(finalPath);
384       finalPath = topDirectory.relativize(finalPath);
385     } else {
386       finalPath = Paths.get(file.getPath()).getFileName();
387     }
388     return finalPath;
389   }

391   private int downloadFiles(DownloadRequest request) throws DownloadException {
392     Path pathPrefix = Paths.get(request.getSampleCode() + File.separator);
393     request
394         .getDataSets()
395         .forEach(
396             dataSetFile -> {
397               try {
398                 int downloadAttempt = 1;
399                 while(downloadAttempt <= request.getMaxNumberOfAttempts()) {
400                   try {
401                     downloadFile(dataSetFile, pathPrefix);
402                     writeCRC32Checksum(dataSetFile, pathPrefix);
403                     return;
404                   } catch (Exception e) {
405                     LOG.error(String.format("Download attempt %d failed.", downloadAttempt));
406                     LOG.error(String.format("Reason: %s", e.getMessage()));
407                     downloadAttempt++;
408                     if (downloadAttempt > request.getMaxNumberOfAttempts()) {
409                       throw new IOException("Maximum number of download attempts reached.");
410                     }
411                   }
412                 }
413               } catch (IOException e) {
414                 String fileName = Paths.get(dataSetFile.getPath()).getFileName().toString();
415                 LOG.error(e);
416                 throw new DownloadException(
417                     "Dataset " + fileName + " could not have been downloaded.");
418               }
419             });
420     return 0;
421   }

423   private void writeCRC32Checksum(DataSetFile dataSetFile, Path pathPrefix) {
424     Path path = Paths.get(pathPrefix.toString(), File.separator,
425         determineFinalPathFromDataset(dataSetFile).toString());
426     checksumReporter.storeChecksum(path, Integer.toHexString(dataSetFile.getChecksumCRC32()));
427   }
