1   package life.qbic.util;
2   
3   import java.text.DateFormat;
4   import java.text.SimpleDateFormat;
5   import java.util.Date;
6   import java.util.TimeZone;
7   import life.qbic.model.units.UnitConverterFactory;
8   import life.qbic.model.units.UnitDisplay;
9   
10  public class ProgressBar {
11  
12    private final int BARSIZE = jline.TerminalFactory.get().getWidth() / 3;
13    private final int MAXFILENAMESIZE = jline.TerminalFactory.get().getWidth() / 3;
14    private float nextProgressJump;
15    private float stepSize;
16    private String fileName;
17    private Long totalFileSize;
18    private Long downloadedSize;
19    private UnitDisplay unitDisplay;
20    private long start;
21    private long remainingTime;
22  
23    public ProgressBar() {}
24  
25    public ProgressBar(String fileName, long totalFileSize) {
26      this.fileName = shortenFileName(fileName);
27      this.totalFileSize = totalFileSize;
28      this.downloadedSize = 0L;
29      this.stepSize = totalFileSize / BARSIZE;
30      this.nextProgressJump = this.stepSize;
31      this.unitDisplay = UnitConverterFactory.determineBestUnitType(totalFileSize);
32      this.start = System.currentTimeMillis();
33      this.remainingTime = 0L;
34    }
35  
36    public void updateProgress(int addDownloadedSize) {
37      this.downloadedSize += (long) addDownloadedSize;
38      checkForJump();
39    }
40  
41    private String shortenFileName(String fullFileName) {
42      String shortName;
43      if (fullFileName.length() > MAXFILENAMESIZE) {
44        shortName = fullFileName.substring(0, MAXFILENAMESIZE - 3) + "...";
45      } else {
46        shortName = fullFileName;
47      }
48      return shortName;
49    }
50  
51    private void checkForJump() {
52      if (this.downloadedSize > this.nextProgressJump) {
53        this.nextProgressJump += this.stepSize;
54        drawProgress();
55      }
56    }
57  
58    private void drawProgress() {
59      System.out.print(
60          String.format("%-" + computeLeftPadding() + "s %s\r", this.fileName, buildProgressBar()));
61    }
62  
63    private int computeLeftPadding() {
64      return MAXFILENAMESIZE + 5;
65    }
66  
67    private String buildProgressBar() {
68      StringBuilder progressBar = new StringBuilder("[");
69      int numberProgressStrings = Math.min((int) (this.downloadedSize / this.stepSize), BARSIZE);
70  
71      double elapsedTimeSeconds = (System.currentTimeMillis() - this.start) / 1000.0;
72  
73      // Download Speed in Mbyte/s
74      double downloadSpeed = this.downloadedSize / (1000000.0 * elapsedTimeSeconds);
75  
76      // Estimate remaining download time
77      long remainingDownloadTime = estimateRemainingTime(downloadSpeed * 1000000.0 / 1000.0);
78      DateFormat dateFormat = new SimpleDateFormat("HH:mm:ss");
79      dateFormat.setTimeZone(TimeZone.getTimeZone("UTC"));
80      String remainingTime = dateFormat.format(new Date(remainingDownloadTime));
81  
82      for (int i = 0; i < numberProgressStrings; i++) {
83        progressBar.append("#");
84      }
85      for (int i = numberProgressStrings; i < BARSIZE; i++) {
86        progressBar.append(" ");
87      }
88  
89      progressBar.append("]\t");
90      progressBar.append(
91          String.format(
92              "%6.02f/%-7.02f%s [%s] (%.02f Mb/s)",
93              unitDisplay.convertBytesToUnit(this.downloadedSize),
94              unitDisplay.convertBytesToUnit(this.totalFileSize),
95              unitDisplay.getUnitType(),
96              remainingTime,
97              downloadSpeed));
98  
99      return progressBar.toString();
100   }
101 
102   /**
103    * Estimates the remaining download time in milliseconds.
104    *
105    * @param downloadSpeed The current download speed in bytes per second
106    * @return The estimated remaining download time in milliseconds
107    */
108   private long estimateRemainingTime(double downloadSpeed) {
109     long remainingFileSize = this.totalFileSize - this.downloadedSize;
110     return (long) (remainingFileSize / downloadSpeed);
111   }
112 }
