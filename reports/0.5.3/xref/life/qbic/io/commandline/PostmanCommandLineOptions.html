1   package life.qbic.io.commandline;
2   
3   import java.nio.file.Path;
4   import java.util.ArrayList;
5   import java.util.List;
6   import picocli.CommandLine.Command;
7   import picocli.CommandLine.Option;
8   import picocli.CommandLine.Parameters;
9   
10  @Command(
11      name = "Postman",
12      footer =
13          "Optional: specify a config file by running postman with '@/path/to/config.txt'. Details can be found in the README.",
14      description =
15          "A client software for dataset downloads from QBiC's data management system openBIS.")
16  public class PostmanCommandLineOptions {
17  
18    @Option(
19        names = {"-u", "--user"},
20        required = true,
21        description = "openBIS user name")
22    public String user;
23  
24    @Option(
25            names = {"-p", "--env-password"},
26            description = "provide the name of an environment variable to read in the password from")
27    public String passwordEnvVariable;
28  
29    // this consumes all parameters that are not labeled!
30    @Parameters(paramLabel = "SAMPLE_ID", description = "one or more QBiC sample ids")
31    public List<String> ids;
32  
33    @Option(
34        names = {"-f", "--file"},
35        description = "a file with line-separated list of QBiC sample ids")
36    public Path filePath;
37  
38    @Option(
39        names = {"-c", "--conserve"},
40        description = "Conserve the file path structure during download")
41    public boolean conservePath;
42  
43    @Option(
44        names = {"-b", "--buffer-size"},
45        description =
46            "dataset download performance can be improved by increasing this value with a multiple of 1024 (default)."
47                + " Only change this if you know what you are doing.")
48    public int bufferMultiplier = 1;
49  
50    @Option(
51        names = {"-s", "--suffix"},
52        description = "returns all files of datasets containing the supplied suffix")
53    public List<String> suffixes = new ArrayList<>();
54  
55    @Option(
56        names = {"-r", "--regex"},
57        description = "returns all files of datasets using your supplied regular expression")
58    public List<String> regexPatterns = new ArrayList<>();
59  
60    @Option(
61        names = {"-t", "--type"},
62        description = "filter for a given openBIS dataset type")
63    public String datasetType = "";
64  
65    @Option(
66        names = {"-dss", "--dss_url"},
67        description = "DataStoreServer URL")
68    public String dss_url = "https://qbis.qbic.uni-tuebingen.de/datastore_server";
69  
70    @Option(
71        names = {"-as", "-as_url"},
72        description = "ApplicationServer URL")
73    public String as_url = "https://qbis.qbic.uni-tuebingen.de/openbis/openbis";
74  
75    @Option(
76        names = {"-h", "--help"},
77        usageHelp = true,
78        description = "display a help message")
79    public boolean helpRequested = false;
80  }
